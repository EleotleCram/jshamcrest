  5 /**
  6  * Asserts that the actual number is greater than the given threshold. Ex: <p>
  7  *
  8  * <pre>
  9  * assertThat(10, greaterThan(5));
 10  * </pre>
 11  *
 12  * @param {number} threshold Threshold number.
 13  * @return {JsHamcrest.SimpleMatcher} 'greaterThan' matcher.
 14  */
 15 JsHamcrest.Matchers.greaterThan = function(threshold) {
 16     return new JsHamcrest.SimpleMatcher({
 17         matches: function(actual) {
 18             return actual > threshold;
 19         },
 20 
 21         describeTo: function(description) {
 22             description.append('greater than ').appendLiteral(threshold);
 23         }
 24     });
 25 };

159 /**
160  * Asserts that the actual number is between a given inclusive range. Ex: <p>
161  * 
162  * <pre>
163  * assertThat(5, between(4).and(7));
164  * </pre>
165  *
166  * @param {number} number Range start.
167  * @return {JsHamcrest.RangeMatcherBuilder} 'between' matcher.
168  */
169 JsHamcrest.Matchers.between = function(number) {
170     return new JsHamcrest.RangeMatcherBuilder({
171         start: number
172     });
173 };

175 /**
176  * Asserts that the actual number is close to the given number, that is, if
177  * the actual number is equal to a number within some range of acceptable error.
178  * Ex: <p>
179  *
180  * <pre>
181  * assertThat(0.5, closeTo(1.0, 0.5));
182  * assertThat(1.0, closeTo(1.0, 0.5));
183  * assertThat(1.5, closeTo(1.0, 0.5));
184  * assertThat(2.0, not(closeTo(1.0, 0.5)));
185  * </pre>
186  *
187  * @param {number} number Number.
188  * @param {number} [delta=0] Acceptable difference range.
189  * @return {JsHamcrest.SimpleMatcher} 'closeTo' matcher.
190  */
191 JsHamcrest.Matchers.closeTo = function(number, delta) {
192     if (!delta) {
193         delta = 0;
194     }
195 
196     return new JsHamcrest.SimpleMatcher({
197         matches: function(actual) {
198             return (Math.abs(actual - number) - delta) <= 0;
199         },
200 
201         describeTo: function(description) {
202             description.append('number within ')
203                   .appendLiteral(delta).append(' of ').appendLiteral(number);
204         }
205     });
206 };

208 /**
209  * Creates a number range matcher builder.
210  * @class Matcher builder that provides an easy way to create matchers for
211  * number ranges.
212  * @constructor
213  * @param {object} param Configuration object.
214  * @param {number} param.start Range start.
215  */
216 JsHamcrest.RangeMatcherBuilder = function(params) {
217     params = params || {};
218 
219     /**
220      * Range start.
221      * @property
222      * @type number
223      * @private
224      */
225     var start = params.start;
226 
227     /**
228      * Finishes to build the range matcher.
229      * @param {number} end Range end.
230      * @return {JsHamcrest.SimpleMatcher} Range matcher.
231      */
232     this.and = function(end) {
233         var greater = end;
234         var lesser = start;
235 
236         if (start > end) {
237             greater = start;
238             lesser = end;
239         }
240 
241         return new JsHamcrest.SimpleMatcher({
242             matches: function(actual) {
243                 return actual >= lesser && actual <= greater;
244             },
245 
246             describeTo: function(description) {
247                 description.append('between ').appendLiteral(lesser)
248                       .append(' and ').appendLiteral(greater);
249             }
250         });
251     }
252 };
