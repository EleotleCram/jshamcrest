  6 /**
  7  * Asserts that the actual object contains the given member (variable or
  8  * function). Ex: <p>
  9  *
 10  * <pre>
 11  * assertThat(myObj, hasMember('name'));
 12  * </pre>
 13  *
 14  * @param {string} memberName Member name.
 15  * @return {JsHamcrest.SimpleMatcher} 'hasMember' matcher.
 16  */
 17 JsHamcrest.Matchers.hasMember = function(memberName) {
 18     return new JsHamcrest.SimpleMatcher({
 19         matches: function(actual) {
 20             try {
 21                 return memberName in actual;
 22             } catch (e) { }
 23             return false;
 24         },
 25 
 26         describeTo: function(description) {
 27             description.append('has member ').appendLiteral(memberName);
 28         }
 29     });
 30 };

 32 /**
 33  * Asserts that the actual object contains the given function. Ex: <p>
 34  *
 35  * <pre>
 36  * assertThat(myObj, hasFunction('getName'));
 37  * </pre>
 38  *
 39  * @param {string} property Property name.
 40  * @return {JsHamcrest.SimpleMatcher} 'hasFunction' matcher.
 41  */
 42 JsHamcrest.Matchers.hasFunction = function(functionName) {
 43     return new JsHamcrest.SimpleMatcher({
 44         matches: function(actual) {
 45             try {
 46                 return functionName in actual && 
 47                         actual[functionName] instanceof Function;
 48             } catch (e) { }
 49             return false;
 50         },
 51 
 52         describeTo: function(description) {
 53             description.append('has function ').appendLiteral(functionName);
 54         }
 55     });
 56 };

 58 /**
 59  * Asserts that the actual object is instance of the given class. Ex: <p>
 60  *
 61  * <pre>
 62  * assertThat(myObj, instanceOf(Array));
 63  * </pre>
 64  *
 65  * @param {function} clazz Constructor function.
 66  * @return {JsHamcrest.SimpleMatcher} 'instanceOf' matcher.
 67  */
 68 JsHamcrest.Matchers.instanceOf = function(clazz) {
 69     return new JsHamcrest.SimpleMatcher({
 70         matches: function(actual) {
 71             return !!(actual instanceof clazz);
 72         },
 73 
 74         describeTo: function(description) {
 75             var className = clazz.name ? clazz.name : 'a class';
 76             description.append('instance of ').append(className);
 77         }
 78     });
 79 };

 81 /**
 82  * Asserts that the actual object is of the specified type. Ex: <p>
 83  *
 84  * <pre>
 85  * assertThat("text", typeOf("string"));
 86  * </pre>
 87  *
 88  * @param {function} typeName Type name.
 89  * @return {JsHamcrest.SimpleMatcher} 'instanceOf' matcher.
 90  */
 91 JsHamcrest.Matchers.typeOf = function(typeName) {
 92     return new JsHamcrest.SimpleMatcher({
 93         matches: function(actual) {
 94             return (typeof actual == typeName);
 95         },
 96 
 97         describeTo: function(description) {
 98             description.append('typeof ').append('"').append(typeName)
 99                     .append('"');
100         }
101     });
102 };

113  */

115     return new JsHamcrest.Matchers.instanceOf(Object);
