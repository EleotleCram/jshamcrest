 12     /**
 13      * Copy all assertion matchers to the given object.
 14      * @param {object} target Target object.
 15      * @private
 16      */
 17     _copyMatchers: function(target) {
 18         var source = JsHamcrest.Matchers;
 19         for (method in source) {
 20             if (!(method in target)) {
 21                 target[method] = source[method];
 22             }
 23         }
 24     },

 26     /**
 27      * JsTestDriver integration. To plug JsHamcrest to JsTestDriver, follow
 28      * JsTestDriver installation and configuration instructions and then
 29      * perform the following changes: <p>
 30      *
 31      * <ol>
 32      *     <li>Let's assume your project root directory have a <em>lib</em>
 33      *     directory to keep your project's dependencies. In this case, copy
 34      *     the <em>jshamcrest.js</em> file to that directory;</li>
 35      *
 36      *     <li>Create a file <em>plugin/jshamcrest-plugin.js</em> in
 37      *     your project root directory and put the following line inside
 38      *     it: <p>
 39      *
 40      *     <pre>
 41      * JsHamcrest.Integration.JsTestDriver();
 42      *     </pre></li>
 43      *
 44      *     <li>Finally, edit the <em>jsTestDriver.conf</em> file as
 45      *     follows: <p>
 46      *     <pre>
 47      * load:
 48      *   - lib/*.js
 49      *   - <source directory>
 50      *   - <test cases directory>
 51      *   - plugin/*.js
 52      *     </pre></li>
 53      * </ol>
 54      *
 55      * That's it! Your test cases should now have access to JsHamcrest
 56      * methods: <p>
 57      *
 58      * <pre>
 59      *     CalculatorTest = TestCase("CalculatorTest");
 60      *
 61      *     CalculatorTest.prototype.testAdd = function() {
 62      *         var calc = new MyCalculator();
 63      *         assertThat(calc.add(2,3), equalTo(5));
 64      *     };
 65      * </pre>
 66      *
 67      * @param {object} params Configuration object.
 68      * @param {object} [params.scope=window] Copies all test matcher functions
 69      * to the given scope.
 70      */
 71     JsTestDriver: function(params) {
 72         params = params ? params : {};
 73         var target = params.scope || window;
 74 
 75         JsHamcrest.Integration._copyMatchers(target);
 76 
 77         /**
 78          * Function called when an assertion fails.
 79          * @ignore
 80          */
 81         var _fail = function(message) {
 82             var exc = new Error(message);
 83             exc.name = 'AssertError';
 84 
 85             // Removes all jshamcrest-related entries from error stack
 86             var stack = exc.stack.split('\n');
 87             var newStack = '';
 88             for (var i = 0; i < stack.length; i++) {
 89                 if (!/jshamcrest*\.js\:/i.test(stack[i])) {
 90                     newStack += stack[i] + '\n';
 91                 }
 92             }
 93             exc.stack = newStack;
 94             throw exc;
 95         };
 96 
 97         /**
 98          * Assertion method exposed to JsTestDriver.
 99          * @ignore
100          */
101         target.assertThat = function (actual, matcher, message) {
102             return JsHamcrest.assertThat(actual, matcher, message, _fail);
103         };
104     },

106     /**
107      * JsUnitTest integration. To plug JsHamcrest to JsUnitTest, follow
108      * JsUnitTest installation and configuration instructions and then edit
109      * the test suite HTML file as follows: <p>
110      *
111      * <pre>
112      *     <!-- JsUnitTest and dependencies -->
113      *     <script type="text/javascript" src="jsunittest.js"></script>
114      *
115      *     <!-- Don't forget to activate JsUnitTest integration -->
116      *     <script type="text/javascript" src="jshamcrest.js"></script>
117      *     <script type="text/javascript">
118      *         JsHamcrest.Integration.JsUnitTest();
119      *     </script>
120      *
121      *     <!-- Some code... -->
122      *
123      *     <script type="text/javascript">
124      *         new Test.Unit.Runner({
125      *             setup: function() {
126      *             },
127      *
128      *             tearDown: function() {
129      *             },
130      *
131      *             testAdd: function() { with(this) {
132      *                 var calc = new MyCalculator();
133      *                 assertThat(calc.add(2,3), equalTo(5));
134      *             }},
135      *
136      *             // More tests here...
137      *         }, {'testLog':'myLog'});
138      *     </script>
139      * </pre>
140      *
141      * @param {object} params Configuration object.
142      * @param {object} [params.scope=Testcase.prototype] Copies all test
143      * matcher functions to the given scope.
144      */
145     JsUnitTest: function(params) {
146         params = params ? params : {};
147         var target = params.scope || JsUnitTest.Unit.Testcase.prototype;
148 
149         JsHamcrest.Integration._copyMatchers(target);
150 
151         /**
152          * Assertion method exposed to JsUnitTest.
153          * @ignore
154          */
155         target.assertThat = function (actual, matcher, message) {
156             var self = this;
157 
158             /**
159              * Function called when an assertion executes successfully.
160              * @ignore
161              */
162             var pass = function() {
163                 self.pass();
164             };
165 
166             /**
167              * Function called when an assertion fails.
168              * @ignore
169              */
170             var fail = function(message) {
171                 self.fail(message);
172             };
173 
174             return JsHamcrest.assertThat(actual, matcher, message, fail, pass);
175         };
176     },

178     /**
179      * YUITest (Yahoo UI) integration. To plug JsHamcrest to YUITest, follow
180      * YUITest installation and configuration instructions and then edit the
181      * test suite HTML file as follows: <p>
182      * 
183      * <pre>
184      *     <!-- YUITest and dependencies -->
185      *     <script type="text/javascript" src="yahoo-dom-event/yahoo-dom-event.js"></script>
186      *     <script type="text/javascript" src="yuilogger/logger.js"></script>
187      *     <script type="text/javascript" src="yuitest/yuitest.js"></script>
188      *
189      *     <!-- Don't forget to activate YUITest integration -->
190      *     <script type="text/javascript" src="jshamcrest.js"></script>
191      *     <script type="text/javascript">
192      *         JsHamcrest.Integration.YUITest();
193      *     </script>
194      *
195      *     <!-- Some code... -->
196      *
197      *     <script type="text/javascript">
198      *         CalculatorTestCase = new YAHOO.tool.TestCase({
199      *             name: "Calculator test case",
200      *
201      *             setUp: function() {
202      *             },
203      *
204      *             teardown: function() {
205      *             },
206      *
207      *             // In YUITest, the assertion method is Assert.that(), not assertThat()!
208      *             // JsHamcrest tries not to screw with the conventions adopted by the
209      *             // unit test framework in use, making the integration feel less
210      *             // intrusive and more natural.
211      *
212      *             testAdd: function() {
213      *                 var calc = new MyCalculator();
214      *                 Assert.that(calc.add(2,3), equalTo(5));
215      *             },
216      *
217      *             // More tests here...
218      *         });
219      *     </script>
220      * </pre>
221      *
222      * @param {object} params Configuration object.
223      * @param {object} [params.scope=window] Copies all test matcher functions
224      * to the given scope.
225      */
226     YUITest: function(params) {
227         params = params ? params : {};
228         var target = params.scope || window;
229 
230         JsHamcrest.Integration._copyMatchers(target);
231         target.Assert = YAHOO.util.Assert;
232 
233         /**
234          * Function called when an assertion fails.
235          * @ignore
236          */
237         var fail = function(message) {
238             YAHOO.util.Assert.fail(message);
239         };
240 
241         /**
242          * Assertion method exposed to YUITest.
243          * @ignore
244          */
245         YAHOO.util.Assert.that = function(actual, matcher, message) {
246             return JsHamcrest.assertThat(actual, matcher, message, fail);
247         };
248     },

272      *     </script>

274      *     <!-- Some code... -->
275      *
276      *     <!-- QUnit and dependencies -->
277      *     <script type="text/javascript" src="testrunner.js"></script>

279      *
280      * @param {object} params Configuration object.
281      * @param {object} [params.scope=window] Copies all test matcher functions
282      * to the given scope.
283      */
284     QUnit: function(params) {
285         params = params ? params : {};
286         var target = params.scope || window;
287 
288         JsHamcrest.Integration._copyMatchers(target);
289 
290         /**
291          * Function called when an assertion executes successfully.
292          * @ignore
293          */
294         var pass = function(message) {
295             QUnit.ok(true, message);
296         };
297 
298         /**
299          * Function called when an assertion fails.
300          * @ignore
301          */
302         var fail = function(message) {
303             QUnit.ok(false, message);
304         };
305 
306         /**
307          * Assertion method exposed to QUnit.
308          * @ignore
309          */
310         target.assertThat = function(actual, matcher, message) {
311             return JsHamcrest.assertThat(actual, matcher, message, fail, pass);
312         };

315     /**
316      * jsUnity integration. To plug JsHamcrest to jsUnity, follow jsUnity
317      * installation and configuration instructions and then edit the test
318      * suite file as follows: <p>
319      *
320      * <pre>
321      *     // Some test suite
322      *     function CalculatorTestSuite() {
323      *         function testA() {
324      *             var calc = new MyCalculator();
325      *             assertThat(calc.add(2,3), equalTo(5));
326      *         }
327      *
328      *         // More tests here...
329      *     }
330      *
331      *     // Don't forget to activate the jsUnity integration
332      *     JsHamcrest.Integration.jsUnity();
333      *
334      *     var results = jsUnity.run(CalculatorTestSuite);
335      * </pre>
336      *
337      * @param {object} params Configuration object.
338      * @param {object} [params.scope=jsUnity.env.defaultScope] Copies all test
339      * matcher functions to the given scope.
340      * @param {object} [params.attachAssertions=false] Whether JsHamcrest
341      * should also copy jsUnity's assertion functions to the given scope.
342      */
343     jsUnity: function(params) {
344         params = params ? params : {};
345         var target = params.scope || jsUnity.env.defaultScope;
346         var assertions = params.attachAssertions || false;
347 
348         JsHamcrest.Integration._copyMatchers(target);
349         if (assertions) {
350             jsUnity.attachAssertions(target);
351         }
352 
353         /**
354          * Function called when an assertion fails.
355          * @ignore
356          */
357         var fail = function(message) {
358             throw message;
359         };
360 
361         /**
362          * Assertion method exposed to jsUnity.
363          * @ignore
364          */
365         target.assertThat = function(actual, matcher, message) {
366             return JsHamcrest.assertThat(actual, matcher, message, fail);
367         };
368     }
