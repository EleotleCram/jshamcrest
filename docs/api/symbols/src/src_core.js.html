 11 /**
 12  * The actual value must be any value considered truth by the JavaScript
 13  * engine. Ex: <p>
 14  *
 15  * <pre>
 16  * assertThat(10, truth());
 17  * assertThat({}, truth());
 18  * assertThat(0, not(truth()));
 19  * assertThat('', not(truth()));
 20  * </pre>
 21  *
 22  * @return {JsHamcrest.SimpleMatcher} 'truth' matcher.
 23  */
 24 JsHamcrest.Matchers.truth = function() {
 25     return new JsHamcrest.SimpleMatcher({
 26         matches: function(actual) {
 27             return actual;
 28         },
 29 
 30         describeTo: function(description) {
 31             description.append('truth');
 32         }
 33     });
 34 };

 36 /**
 37  * Delegate-only matcher frequently used to improve readability. Ex: <p>
 38  *
 39  * <pre>
 40  * assertThat(10, is(10));
 41  * assertThat(10, is(equalTo(10)));
 42  * </pre>
 43  *
 44  * @param {object} matcher Delegate matcher.
 45  * @return {JsHamcrest.SimpleMatcher} 'is' matcher.
 46  */
 47 JsHamcrest.Matchers.is = function(matcher) {
 48     // Uses 'equalTo' matcher if the given object is not a matcher
 49     if (!JsHamcrest.isMatcher(matcher)) {
 50         matcher = JsHamcrest.Matchers.equalTo(matcher);
 51     }
 52 
 53     return new JsHamcrest.SimpleMatcher({
 54         matches: function(actual) {
 55             return matcher.matches(actual);
 56         },
 57 
 58         describeTo: function(description) {
 59             description.append('is ').appendDescriptionOf(matcher);
 60         }
 61     });
 62 };

 92 /**
 93  * The actual value must be equal to the given value to be successful.
 94  * Ex: <p>
 95  *
 96  * <pre>
 97  * assertThat(10, equalTo('10'));
 98  * </pre>
 99  *
100  * @param {object} expected value.
101  * @return {JsHamcrest.SimpleMatcher} 'equalTo' matcher.
102  */
103 JsHamcrest.Matchers.equalTo = function(expected) {
104     return new JsHamcrest.SimpleMatcher({
105         matches: function(actual) {
106             if (expected instanceof Array || actual instanceof Array) {
107                 return JsHamcrest.isArraysEqual(expected, actual);
108             }
109             return actual == expected;
110         },
111 
112         describeTo: function(description) {
113             description.append('equal to ').appendLiteral(expected);
114         }
115     });
116 };

184 /**
185  * The actual value is a function and, when invoked, it should thrown an
186  * exception with the given name to be successful. Ex: <p>
187  *
188  * <pre>
189  * var MyException = function(message) {
190  *   this.name = 'MyException';
191  *   this.message = message;
192  * };
193  * 
194  * var myFunction = function() {
195  *   // Do something dangerous...
196  *   throw new MyException('Unexpected error');
197  * }
198  *
199  * assertThat(myFunction, raises('MyException'));
200  * </pre>
201  *
202  * @param {string} exceptionName Name of the expected exception.
203  * @return {JsHamcrest.SimpleMatcher} 'raises' matcher
204  */
205 JsHamcrest.Matchers.raises = function(exceptionName) {
206     return new JsHamcrest.SimpleMatcher({
207         matches: function(actualFunction) {
208             try {
209                 actualFunction();
210             } catch (e) {
211                 if (e.name == exceptionName) {
212                     return true;
213                 } else {
214                     throw e;
215                 }
216             }
217             return false;
218         },
219 
220         describeTo: function(description) {
221             description.append('raises ').append(exceptionName);
222         }
223     });
224 }

268 /**
269  * All the given values or matchers should match the actual value to be
270  * sucessful. This matcher behaves pretty much like the JavaScript &&
271  * operator (short-circuiting). Ex: <p>
272  *
273  * <pre>
274  * assertThat(5, allOf([greaterThan(0), lessThan(10)]));
275  * assertThat(5, allOf([5, lessThan(10)]));
276  * assertThat(5, allOf(greaterThan(0), lessThan(10)));
277  * assertThat(5, allOf(5, lessThan(10)));
278  * </pre>
279  *
280  * @param {array} arguments List of delegate matchers.
281  * @return {JsHamcrest.SimpleMatcher} 'allOf' matcher.
282  */
283 JsHamcrest.Matchers.allOf = function() {
284     var args = arguments;
285     if (args[0] instanceof Array) {
286         args = args[0];
287     }
288     return new JsHamcrest.SimpleMatcher({
289         matches: function(actual) {
290             for (var i = 0; i < args.length; i++) {
291                 var matcher = args[i];
292                 if (!JsHamcrest.isMatcher(matcher)) {
293                     matcher = JsHamcrest.Matchers.equalTo(matcher);
294                 }
295                 if (!matcher.matches(actual)) {
296                     return false;
297                 }
298             }
299             return true;
300         },
301 
302         describeTo: function(description) {
303             description.appendList('(', ' and ', ')', args);
304         }
305     });
306 };

308 /**
309  * At least one of the given matchers should match the actual value to be
310  * sucessful. This matcher behaves pretty much like the JavaScript ||
311  * operator (short-circuiting). Ex: <p>
312  *
313  * <pre>
314  * assertThat(5, not(anyOf(lessThan(0), greaterThan(100))));
315  * </pre>
316  *
317  * @param {array} arguments List of delegate matchers.
318  * @return {JsHamcrest.SimpleMatcher} 'anyOf' matcher.
319  */
320 JsHamcrest.Matchers.anyOf = function() {
321     var args = arguments;
322     if (args[0] instanceof Array) {
323         args = args[0];
324     }
325     return new JsHamcrest.SimpleMatcher({
326         matches: function(actual) {
327             for (var i = 0; i < args.length; i++) {
328                 var matcher = args[i];
329                 if (!JsHamcrest.isMatcher(matcher)) {
330                     matcher = JsHamcrest.Matchers.equalTo(matcher);
331                 }
332                 if (matcher.matches(actual)) {
333                     return true;
334                 }
335             }
336             return false;
337         },
338 
339         describeTo: function(description) {
340             description.appendList('(', ' or ', ')', args);
341         }
342     });
343 };
