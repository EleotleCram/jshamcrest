 22 JsHamcrest = {
 23     /**
 24      * Library version.

 28     /**
 29      * Assert method that is capable of handling matchers. If the given matcher
 30      * fails, this method registers a failed/error'd assertion with the unit
 31      * test framework being used. Ex: <p>
 32      *
 33      * <pre>
 34      * // Asserts that something is equal to x
 35      * assertThat(something, equalTo(x));
 36      * assertThat(something, equalTo(x), "Some description text");
 37      *
 38      * // Same here
 39      * assertThat(something, x);
 40      * assertThat(something, x, "Some description text");
 41      *
 42      * // Asserts that something evaluates to some value considered truth
 43      * assertThat(something);
 44      * </pre>
 45      *
 46      * @param {object} actual Actual value under test.
 47      * @param {object} matcher Matcher to assert the correctness of the actual
 48      * value.
 49      * @param {string} message Message that describes the assertion, if
 50      * necessary.
 51      * @param {function} fail Function to be called when the assertion fails.
 52      * @param {function} pass Function to be called when the assertion
 53      * succeeds.
 54      * @return {JsHamcrest.Description} Test result description.
 55      */
 56     assertThat: function(actual, matcher, message, fail, pass) {
 57         var description = new JsHamcrest.Description();
 58         var matchers = JsHamcrest.Matchers;
 59 
 60         // Actual value must be any value considered non-null by JavaScript
 61         if (matcher == null) {
 62             matcher = matchers.truth();
 63         }
 64 
 65         // Creates a 'equalTo' matcher if 'matcher' is not a valid matcher
 66         if (!JsHamcrest.isMatcher(matcher)) {
 67             matcher = matchers.equalTo(matcher);
 68         }
 69 
 70         if (message) {
 71             description.append(message).append('. ');
 72         }
 73         description.append('Expected ');
 74         matcher.describeTo(description);
 75 
 76         if (!matcher.matches(actual)) {
 77             description.append(' but was ');
 78             matcher.describeValueTo(actual, description);
 79             fail(description.get());
 80         } else {
 81             description.append(': Success');
 82             if (pass) {
 83                 pass(description.get());
 84             }
 85         }
 86         return description;
 87     },

 89     /**
 90      * Returns whether the given object is a matcher.
 91      * @param {object} obj Object.
 92      * @return {boolean} Whether the given object is a matcher.
 93      */
 94     isMatcher: function(obj) {
 95         return obj instanceof JsHamcrest.SimpleMatcher;
 96     },
 97 
 98     /**
 99      * Returns whether the given arrays are equivalent.
100      * @param {array} array Array.
101      * @param {array} anotherArray Another array.
102      * @return {booelan} Whether the given arrays are equivalent.
103      */
104     isArraysEqual: function(array, anotherArray) {
105         if (array instanceof Array || anotherArray instanceof Array) {
106             if (array.length != anotherArray.length) {
107                 return false;
108             }
109 
110             for (var i = 0; i < array.length; i++) {
111                 var a = array[i];
112                 var b = anotherArray[i];
113 
114                 if (a instanceof Array || b instanceof Array) {
115                     return JsHamcrest.isArraysEqual(a, b);
116                 } else if (a != b) {
117                     return false;
118                 }
119             }
120             return true;
121         } else {
122             return array == anotherArray;
123         }
124     },

126     /**
127      * Creates a simple matcher.
128      * @class Builds a matcher object that uses external functions provided
129      * by the caller in order to define the current matching logic.
130      * @constructor
131      * @param {object} params Configuration object.
132      * @param {function} params.matches Matcher logic.
133      * @param {function} params.describeTo Self description logic. This
134      * function is used to create textual descriptions from matcher objects.
135      * @param {function} [params.describeValueTo] This function is used to
136      * describe the actual value of a test assertion. If not provided the
137      * actual value will be described as a literal.
138      */
139     SimpleMatcher: function(params) {
140         params = params || {};
141 
142         /**
143          * Checks if this matcher matches the actual value.
144          * @function
145          * @param {object} actual Actual value.
146          * @return {boolean} If they match or not.
147          */
148         this.matches = params.matches;
149 
150         /**
151          * Describes this matcher's tasks to the given descriptor.
152          * @function
153          * @param {object} descriptor Descriptor.
154          */
155         this.describeTo = params.describeTo;
156 
157         // Replace the function to describe the actual value
158         if (params.describeValueTo) {
159             this.describeValueTo = params.describeValueTo;
160         }
161     },
162 
163     /**
164      * Creates a combinable matcher.
165      * @class Matcher that provides an easy way to wrap several matchers into
166      * one.
167      * @param {object} params Configuration object.
168      * @param {function} params.matches Matcher logic.
169      * @param {function} params.describeTo Self description logic. This
170      * function is used to create textual descriptions from matcher objects.
171      */
172     CombinableMatcher: function(params) {
173         // Call superclass' constructor
174         JsHamcrest.SimpleMatcher.apply(this, arguments);
175 
176         params = params || {};
177 
178         /**
179          * Wraps this matcher with the given one in such a way that both
180          * matchers must match the actual value to be successful.
181          * @param {object} anotherMatcher Another matcher.
182          * @return {JsHamcrest.CombinableMatcher} Combinable matcher.
183          */
184         this.and = function(anotherMatcher) {
185             var all = JsHamcrest.Matchers.allOf(this, anotherMatcher);
186             return new JsHamcrest.CombinableMatcher({
187                 matches: all.matches,
188 
189                 describeTo: function(description) {
190                     description.appendDescriptionOf(all);
191                 }
192             });
193         };
194 
195         /**
196          * Wraps this matcher with the given one in such a way that at least
197          * one of the matchers must match the actual value to be successful.
198          * @param {object} anotherMatcher Another matcher.
199          * @return {JsHamcrest.CombinableMatcher} Combinable matcher.
200          */
201         this.or = function(anotherMatcher) {
202             var any = JsHamcrest.Matchers.anyOf(this, anotherMatcher);
203             return new JsHamcrest.CombinableMatcher({
204                 matches: any.matches,
205 
206                 describeTo: function(description) {
207                     description.appendDescriptionOf(any);
208                 }
209             });
210         };
211     },
212 
213     /**
214      * Creates a description.
215      * @class Description is the object that builds assertion error messages.
216      * @constructor
217      */
218     Description: function() {
219         /**
220          * Current content of this description.
221          * @property
222          * @type string
223          * @private
224          */
225         var value = '';
226 
227         /**
228          * Gets the current content of this description.
229          * @return {string} Current content of this description.
230          */
231         this.get = function() {
232             return value;
233         }
234 
235         /**
236          * Appends the description a self describing object to this
237          * description.
238          * @param {object} selfDescribing Any object that have a
239          * <code>describeTo</code> method that accepts a description object as
240          * argument.
241          * @return {JsHamcrest.Description} this.
242          */
243         this.appendDescriptionOf = function(selfDescribing) {
244             if (selfDescribing) {
245                 selfDescribing.describeTo(this);
246             }
247             return this;
248         };
249 
250         /**
251          * Appends a text to this description.
252          * @param {string} text Text to append.
253          * @return {JsHamcrest.Description} this.
254          */
255         this.append = function(text) {
256             if (text != null) {
257                 value += text;
258             }
259             return this;
260         };
261 
262         /**
263          * Appends a JavaScript language's literals to this description.
264          * @param {object} literal Literal to append.
265          * @return {JsHamcrest.Description} this.
266          */
267         this.appendLiteral = function(literal) {
268             if (literal === undefined) {
269                 this.append('undefined');
270             } else if (literal === null) {
271                 this.append('null');
272             } else if (literal instanceof Array) {
273                 this.appendValueList('[', ', ', ']', literal);
274             } else if (typeof literal == 'string') {
275                 this.append('"' + literal + '"');
276             } else if (literal instanceof Function) {
277                 this.append('Function');
278             } else {
279                 this.append(literal);
280             }
281             return this;
282         }
283 
284         /**
285          * Appends a list of values to this description.
286          * @param {string} start Start string.
287          * @param {string} separator Separator string.
288          * @param {string} end End string.
289          * @param {array} list List of values.
290          * @return {JsHamcrest.Description} this.
291          */
292         this.appendValueList = function(start, separator, end, list) {
293             this.append(start);
294             for (var i = 0; i < list.length; i++) {
295                 if (i > 0) {
296                     this.append(separator);
297                 }
298                 this.appendLiteral(list[i]);
299             }
300             this.append(end);
301             return this;
302         };
303 
304         /**
305          * Appends a list of self describing objects to this description.
306          * @param {string} start Start string.
307          * @param {string} separator Separator string.
308          * @param {string} end End string.
309          * @param {array} list List of self describing objects. These objects
310          * must that have a <code>describeTo</code> method that accepts a
311          * description object as argument.
312          * @return {JsHamcrest.Description} this.
313          */
314         this.appendList = function(start, separator, end, list) {
315             this.append(start);
316             for (var i = 0; i < list.length; i++) {
317                 if (i > 0) {
318                     this.append(separator);
319                 }
320                 this.appendDescriptionOf(list[i]);
321             }
322             this.append(end);
323             return this;
324         };
325     }
326 };

329 /**
330  * Describes the actual value to the given descriptor.
331  * This method is optional and, if it's not present,
332  * the actual value will be described as a JavaScript
333  * literal.
334  * @param {object} actual Actual value.
335  * @param {object} descriptor Descriptor.
336  */
337 JsHamcrest.SimpleMatcher.prototype.describeValueTo = function(actual, description) {
338     description.appendLiteral(actual);
339 };
