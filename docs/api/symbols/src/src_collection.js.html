  5 /**
  6  * The actual value should be an array and it must contain at least one value
  7  * that matches the given value or matcher to be successful. Ex: <p>
  8  *
  9  * <pre>
 10  * assertThat([1,2,3], hasItem(3));
 11  * assertThat([1,2,3], hasItem(equalTo(3)));
 12  * </pre>
 13  *
 14  * @param {array} matcher Number or matcher.
 15  * @return {JsHamcrest.SimpleMatcher} 'hasItem' matcher.
 16  */
 17 JsHamcrest.Matchers.hasItem = function(matcher) {
 18     // Uses 'equalTo' matcher if the given object is not a matcher
 19     if (!JsHamcrest.isMatcher(matcher)) {
 20         matcher = JsHamcrest.Matchers.equalTo(matcher);
 21     }
 22 
 23     return new JsHamcrest.SimpleMatcher({
 24         matches: function(actual) {
 25             // Should be an array
 26             if (!(actual instanceof Array)) {
 27                 return false;
 28             }
 29 
 30             for (var i = 0; i < actual.length; i++) {
 31                 if (matcher.matches(actual[i])) {
 32                     return true;
 33                 }
 34             }
 35             return false;
 36         },
 37 
 38         describeTo: function(description) {
 39             description.append('array contains item ')
 40                     .appendDescriptionOf(matcher);
 41         }
 42     });
 43 };

 45 /**
 46  * The actual value should be an array and the given values or matchers must
 47  * match at least one item to be sucessful. Ex: <p>
 48  *
 49  * <pre>
 50  * assertThat([1,2,3], hasItems(2, 3));
 51  * assertThat([1,2,3], hasItems(greaterThan(2)));
 52  * </pre>
 53  *
 54  * @param {object...} arguments Values or matchers.
 55  * @return {JsHamcrest.SimpleMatcher} 'hasItems' matcher.
 56  */
 57 JsHamcrest.Matchers.hasItems = function() {
 58     var items = [];
 59     for (var i = 0; i < arguments.length; i++) {
 60         items.push(JsHamcrest.Matchers.hasItem(arguments[i]));
 61     }
 62     return JsHamcrest.Matchers.allOf(items);
 63 };

 65 /**
 66  * The actual value should be an array and the given value or matcher must
 67  * match all items to be successful. Ex: <p>
 68  *
 69  * <pre>
 70  * assertThat([1,2,3], everyItem(greaterThan(0)));
 71  * </pre>
 72  *
 73  * @param {matcher} matcher Value or matcher.
 74  * @return {JsHamcrest.SimpleMatcher} 'everyItem' matcher.
 75  */
 76 JsHamcrest.Matchers.everyItem = function(matcher) {
 77     // Uses 'equalTo' matcher if the given object is not a matcher
 78     if (!JsHamcrest.isMatcher(matcher)) {
 79         matcher = JsHamcrest.Matchers.equalTo(matcher);
 80     }
 81 
 82     return new JsHamcrest.SimpleMatcher({
 83         matches: function(actual) {
 84             // Should be an array
 85             if (!(actual instanceof Array)) {
 86                 return false;
 87             }
 88 
 89             for (var i = 0; i < actual.length; i++) {
 90                 if (!matcher.matches(actual[i])) {
 91                     return false;
 92                 }
 93             }
 94             return true;
 95         },
 96 
 97         describeTo: function(description) {
 98             description.append('every item ').appendDescriptionOf(matcher);
 99         }
100     });
101 };

103 /**
104  * The given array must contain the actual value to be successful. Ex: <p>
105  * 
106  * <pre>
107  * assertThat(1, isIn([1,2,3]));
108  * assertThat(1, isIn(1,2,3));
109  * </pre>
110  *
111  * @return {JsHamcrest.SimpleMatcher} 'isIn' matcher.
112  */
113 JsHamcrest.Matchers.isIn = function() {
114     var equalTo = JsHamcrest.Matchers.equalTo;
115 
116     var args = arguments;
117     if (args[0] instanceof Array) {
118         args = args[0];
119     }
120 
121     return new JsHamcrest.SimpleMatcher({
122         matches: function(actual) {
123             for (var i = 0; i < args.length; i++) {
124                 if (equalTo(args[i]).matches(actual)) {
125                     return true;
126                 }
127             }
128             return false;
129         },
130 
131         describeTo: function(description) {
132             description.append('one of ').appendLiteral(args);
133         }
134     });
135 };

137 /**
138  * The given array must contain the actual value to be successful. This is an
139  * alias to 'isIn' matcher. Ex: <p>
140  * <pre>
141  * assertThat(1, oneOf([1,2,3]));
142  * assertThat(1, oneOf(1,2,3));
143  * </pre>
144  *
145  * @function
146  * @return {JsHamcrest.SimpleMatcher} 'oneOf' matcher.
147  */
148 JsHamcrest.Matchers.oneOf = JsHamcrest.Matchers.isIn;

150 /**
151  * The actual value should be an array and it must be empty to be sucessful.
152  * Ex: <p>
153  *
154  * <pre>
155  * assertThat([], empty());
156  * </pre>
157  *
158  * @return {JsHamcrest.SimpleMatcher} 'empty' matcher.
159  */
160 JsHamcrest.Matchers.empty = function() {
161     return new JsHamcrest.SimpleMatcher({
162         matches: function(actual) {
163             return actual instanceof Array && actual.length == 0;
164         },
165 
166         describeTo: function(description) {
167             description.append('empty');
168         }
169     });
170 };

172 /**
173  * The actual value should be an array and its size must match the given value
174  * or matcher to be sucessful. Ex: <p>
175  *
176  * <pre>
177  * assertThat([1,2,3], hasSize(3));
178  * assertThat([1,2,3], hasSize(lessThan(5)));
179  * </pre>
180  *
181  * @param {object} matcher Number or matcher.
182  * @return {JsHamcrest.SimpleMatcher} 'hasSize' matcher.
183  */
184 JsHamcrest.Matchers.hasSize = function(matcher) {
185     // Uses 'equalTo' matcher if the given object is not a matcher
186     if (!JsHamcrest.isMatcher(matcher)) {
187         matcher = JsHamcrest.Matchers.equalTo(matcher);
188     }
189 
190     return new JsHamcrest.SimpleMatcher({
191         matches: function(actual) {
192             return actual instanceof Array && matcher.matches(actual.length);
193         },
194 
195         describeTo: function(description) {
196             description.append('has size ').appendDescriptionOf(matcher);
197         },
198 
199         describeValueTo: function(actual, description) {
200             if (actual instanceof Array) {
201                 description.append(actual.length);
202             } else {
203                 description.appendLiteral(actual);
204             }
205         }
206     });
207 };
